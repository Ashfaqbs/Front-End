import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.GetMapping;

@Controller
public class StatusController {

    @GetMapping("/checkAllStatus")
    public String checkAllStatus(Model model) {
        String[] applications = {"spider", "planit", "ccs"};
        String[] environments = {"sit", "uat", "perf", "perf2", "erf3", "ddte", "mve", "mve2", "uat2", "prod"};

        // Store results for each combination
        boolean[][] results = new boolean[applications.length][environments.length];

        // Loop through applications and environments
        for (int i = 0; i < applications.length; i++) {
            for (int j = 0; j < environments.length; j++) {
                String url = buildUrl(applications[i], environments[j]);
                results[i][j] = checkUrl(url);
            }
        }

        // Set model attributes for Thymeleaf template
        model.addAttribute("applications", applications);
        model.addAttribute("environments", environments);
        model.addAttribute("results", results);

        return "responseAll";
    }

    private String buildUrl(String application, String environment) {
