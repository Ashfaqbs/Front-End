Attaching Multiple Handlers to a Promise



        let p1 = new Promise((resolve, reject) => {

            setTimeout(() => {
                resolve(1)
                console.log('Hey im resolved')

            }, 2000)


        })


These handlers run independently

        //Handlers 1
        p1.then((data) => { console.log(data + "handler1") })
        //Handlers 2
        p1.then((data) => { console.log(data + "handler2") })
        //Handlers 3
        p1.then((data) => { console.log(data + "handler3") })
