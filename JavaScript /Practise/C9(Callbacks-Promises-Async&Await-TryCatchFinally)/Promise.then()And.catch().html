Promise .then() and .catch()

 
    let promise1 = new Promise((resolve, reject) => {

        // Logic 
        setTimeout(() => {
            console.log('Hello promise is fulfilled  in 3s')
            resolve(true)
        }, 3000);

    })



    let promise2 = new Promise((resolve, reject) => {

        // Logic 
        setTimeout(() => {
            console.log('Hello promise is rejected')
            reject(false)
        }, 3000);

    })

    //only handling successful scenario
    promise1.then(
        (value) => { console.log(value) }
    )

    //handling failure  scenarios by catch if we dont handle this errors are seen in console 
    promise2.catch((error) => { console.log(error) })

    //handling successful and failure scenario by then 
    promise2.then((data) => { console.log(data) }, 
    (error) => { console.log(error) })
