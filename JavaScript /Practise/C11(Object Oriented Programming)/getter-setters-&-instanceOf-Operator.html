Instanceof Operator

the instanceof Operator allows to check wether an object belongs to a certain class the syntax is : obj>instanceOf class > it returns true if the object belongs to the class or any other class inheriting from it console.log(user1 instanceof User ) // o/p true 
        
            
                class User {

                    id;
                    name;
                    school;
        
                    gotoSchool() {
                        console.log('going to school')
                    }
                    getid() {
        
                        console.log(this.id)
                        return this.id;
                    }
        
                    getname() {
        
                        console.log(this.name)
                        return this.name;
                    }
                    getschool() {
        
                        console.log(this.school)
                        return this.school;
                    }
        
        
                    setname(name) {
                        this.name = name;
        
        
                    }
                    setSchool(sch) {
                        this.school = sch;
        
        
                    } setId(ID) {
                        this.id = ID;
                    }
        
        
                }
        
        
        
                let user1 = new User()
                user1.setId(1)
                user1.setname('sonu')
                user1.setSchool('abc school')
        
        
                console.log(user1.getid() +
                    user1.getname() +
                    user1.getschool())
