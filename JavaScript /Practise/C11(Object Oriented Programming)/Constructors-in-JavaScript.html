Constructors in JavaScript




    

The constructor
the constructor() is called automatically by new keyword , so we can initialize the object there , if we dont provide our own constructor a default constructor will be provided




class User {

    id;
    name;
    place;
    constructor(id, name, place) {
        this.id = id;
        this.name = name;
        this.place = place;
        console.log('calling constructor')

    }

    add(x, y) {
        console.log(x + y)
        return x + y;
    }


    sub(a, b) {
        console.log(a - b)
        return (a - b);
    }

}




let user1 = new User(1, "Sonu", "Bangalore")
let user2 = new User(2, "Monu", "Bangalore")
let user3 = new User(3, "tony", "Bangalore")
