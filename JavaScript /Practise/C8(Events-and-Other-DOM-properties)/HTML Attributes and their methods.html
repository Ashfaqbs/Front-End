Hey this is a span tag 


    let spn = document.getElementById('sp')
    console.log(spn.getAttribute('class'))//this returns the attribute which exists for the tag, as we have provided the class as card so o/p is card 
    console.log(spn.hasAttribute('class'))//check's wether the span tag has class attribute in it o/p true as we have given class to span tag 
    // true
    console.log(spn.hasAttribute('name'))//check's wether the span tag has name attribute in it o/p false as we have not given name to span tag 

    console.log(spn.removeAttribute('class'))//removes the class attribute in the span tag


    spn.setAttribute('height', '200px')//we can add a new attribute and define the value to it

    console.log(spn.hasAttribute('style'))//check's wether the span tag has style attribute in it o/p false as we have not added style to span tag 
    // false

    console.log(spn.attributes)//returns all the attribute of the element
